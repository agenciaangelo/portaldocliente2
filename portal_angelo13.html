        // Cole no Apps Script (script.google.com) function doPost(e){ try{ var p=JSON.parse(e.postData.contents); if(p.action==='uploadArquivo') return uploadToDrive(p); if(p.action==='novaDemanda') return registrarDemanda(p); return resp({ok:false,msg:'Ação desconhecida'}); }catch(err){return resp({ok:false,msg:err.toString()});} } function doGet(e){ var p=e.parameter; if(p.action==='login') return checkLogin(p.user,p.pass); return resp({ok:true,msg:'API ativa'}); } function uploadToDrive(payload){ var folder; try{folder=DriveApp.getFolderById(payload.folderId);} catch(err){return resp({ok:false,msg:'Pasta não encontrada'});} var bytes=Utilities.base64Decode(payload.data); var blob=Utilities.newBlob(bytes,payload.mimeType||'application/octet-stream',payload.fileName); var file=folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK,DriveApp.Permission.VIEW); return resp({ok:true,fileId:file.getId()}); } function checkLogin(user,pass){ return resp({ok:false}); } function registrarDemanda(p){ var ss=SpreadsheetApp.openById('SEU_ID'); var sh=ss.getSheetByName('Demandas')||ss.insertSheet('Demandas'); sh.appendRow([new Date(),p.cliente,p.titulo,p.tipo,p.prazo,p.prioridade,p.desc,'Novo']); return resp({ok:true}); } function resp(obj){ return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
